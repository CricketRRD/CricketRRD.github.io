From: Grimshaw Stuart <stuart.grimshaw@blackburn.gov.uk>

Here is my first attempt at a defaults file. I would appreciate *any*
comments you have to make, and I hope someone else also finds it
usefull.

I'm not sure if the scale is working correctly for the number of pages
printed, but it seems to be....

<Defaults>

# Cricket 'Defaults' file for HP Laserjet 4000 printers
# Created 05/01/2000 Stuart M. Grimshaw

# HP Laserjet 4000 OID's
OID hplj_totalpagecount 1.3.6.1.2.1.43.10.2.1.4.1.1
OID hplj_pspagecount 1.3.6.1.4.1.11.2.3.9.4.2.1.3.3.4.5.0
OID hplj_pclpagecount 1.3.6.1.4.1.11.2.3.9.4.2.1.3.3.3.5.0
OID hplj_storageused 1.3.6.1.4.1.11.2.3.9.4.2.3.2.3.1.6.1

# HP Laserjet 6p OID
OID lj6p_totalpagecount 1.3.6.1.4.1.11.2.3.9.4.2.2.10.2.1.4.1.1

# HP Deskjet 1120c (A3)
OID 1120c_totalpagecount 1.3.6.1.4.1.11.2.3.9.4.2.1.4.1.2.5.0

target --default--
        snmp-host = %printer%

# ---------------------------------------------
# HP Laserjet 4000 Datasources

datasource hplj_totalpagecount
        ds-source = snmp://%snmp%/hplj_totalpagecount
        rrd-ds-type = COUNTER

datasource hplj_pspagecount
        ds-source = snmp://%snmp%/hplj_pspagecount
        rrd-ds-type = COUNTER

datasource hplj_pclpagecount
        ds-source = snmp://%snmp%/hplj_pclpagecount
        rrd-ds-type = COUNTER

datasource hplj_storageused
        ds-source = snmp://%snmp%/hplj_storageused
        rrd-ds-type = GAUGE

# ------------------------------------------
# HP Laserjet 6p datasource

datasource hp6p_totalpagecount
        ds-source = snmp://%snmp%/hp6p_totalpagecount
        rrd-ds-type = COUNTER

# ------------------------------------------
# HP Deskjet 1120c datasource

datasource 1120c_totalpagecount
        ds-source = snmp://%snmp%/1120c_totalpagecount
        rrd-ds-type = COUNTER

# Graph defaults
graph --default--
        si-units = 0
        precision = 0
        draw-as = AREA
        color = orange

# HP Laserjet 4000 Graph defaults
graph hplj_totalpagecount
        legend = "Pages Printed"
        y-axis = "Pages"
        scale = 300,*

graph hplj_pspagecount
        y-axis = "PostScript Pages"
        scale = 300,*

graph hplj_pclpagecount
        color = blue
        draw-as LINE
        y-axis = "PCL Pages"
        scale = 300,*

graph hplj_storageused
        color = dark-green
        y-axis = "Storage Used"
        si-units = true
        precision = 2
        bytes = true

# ------------------------------------------
# HP Laserjet 6p Graph settings

graph hp6p_totalpagecount
        y-axis = "pages"
        scale = 300,*

# ------------------------------------------
# HP Deskjet 1120c graph settings

graph 1120c_totalpagecount
        y-axis = "Pages"
        scale = 300,*

# ------------------------------------------
# Target Types

targetType hp-lj4000
        ds = "hplj_totalpagecount, hplj_pspagecount, hplj_pclpagecount,
hplj_storageused"
        view = "Total_Pages: hplj_totalpagecount, By_PageType: hplj_pspagecount
hplj_pclpagecount, Memory: hplj_storageused"

targetType hp-dj6p
        ds = "hp6p_totalpagecount"
        view = "Pages: hp6p_totalpagecount"

targetType hp_dj1120c
        ds = "1120c_totalpagecount"
        view = "Pages: 1120c_totalpagecount"

</Defaults>

<printers>

target admin_main
        target-type = hp-lj4000
        printer = 10.18.2.2
        short-desc = "Main admin printer, 2nd floor office"

</printers>
