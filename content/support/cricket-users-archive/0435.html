Try this guy out, pretty easy, list-interfaces should do it for ya.
target --default--
        router = router.domain.edu
        directory-desc = "Cisco 4700 Router"

target Ethernet0
        interface-name = Ethernet0

target Ethernet1
        interface-name = Ethernet1

target Serial0
        interface-name = Serial0

target Serial1
        interface-name = Serial1

target Serial0.1
        interface-name = Serial0.1
        target-type = sub-interface

target Serial1.1
        interface-name = Serial1.1
        target-type = sub-interface

target Serial1.2
        interface-name = Serial1.2
        target-type = sub-interface

target Serial1.3
        interface-name = Serial1.3
        target-type = sub-interface
